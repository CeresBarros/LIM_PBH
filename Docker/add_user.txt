## To add a new user (including to RStudio Server):

## enter container
docker exec -it <containerID> bash

## add users and set password
useradd <username>
passwd <username>

## make home dir if not there
mkdir /home/<username>

## change ownership of homedir
chown -R <username>:<username> /home/<username>

## change ownership of scratch
chown -R <username>:<username> /mnt/scratch/<username>

## Changing user/group ID to match a similar external (to Docker) user's 
## outside container check the user's/group ID
id -u <username>
groups <username>

## In docker container, re-check user ID and change it to match external user ID 
usermod -u <new_uid> <username>

## Do the same for group ID -- usualy it is the same as user ID, but check groups first
groups <username>
groupmod -g <new_gid> <groupname>

## Change file permission in user's home to match new ID
chown -R <new_uid>:<new_gid> /home/<username>
